# Definition for singly-linked list.
# class ListNode:
#     def __init__(self, val=0, next=None):
#         self.val = val
#         self.next = next
class Solution:
    def pairSum(self, head: Optional[ListNode]) -> int:
        fast = slow = head
        while fast and fast.next:
            fast = fast.next.next
            slow = slow.next

        cur = slow
        prev = None
        while cur:
            temp = cur.next
            cur.next = prev
            prev = cur 
            cur = temp
        
        mx = 0
        while prev and head:
            mx = max(mx, prev.val + head.val)
            prev = prev.next
            head = head.next
        return mx
