class Solution:
    def smallestNumber(self, num: int) -> int:
        res = sorted(list(str(num)))
        print(res)
        i = 0
        while i < len(res) and res[i] == "0":
            i += 1
        if i < len(res) and res[i - 1] == "0":
            print("YES", i)
            res[i], res[0] = res[0], res[i]

        return int("".join(res if res[0] != "-" else [res[0]] + res[:0:-1]) )
