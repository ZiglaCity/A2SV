# Definition for singly-linked list.
# class ListNode:
#     def __init__(self, val=0, next=None):
#         self.val = val
#         self.next = next
class Solution:
    def swapPairs(self, head: Optional[ListNode]) -> Optional[ListNode]:
        
        cur = head
        if cur and cur.next:
            head = cur.next

        prev = None
        while cur and cur.next:
            temp = cur.next.next
            cur.next.next = cur
            if prev:
                prev.next = cur.next
            cur.next = temp
            prev = cur
            cur = temp
        return head
