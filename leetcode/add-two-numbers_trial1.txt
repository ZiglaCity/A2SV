# Definition for singly-linked list.
# class ListNode:
#     def __init__(self, val=0, next=None):
#         self.val = val
#         self.next = next
class Solution:
    def addTwoNumbers(self, l1: Optional[ListNode], l2: Optional[ListNode]) -> Optional[ListNode]:

        carry = 0
        cur = head = ListNode(-1)
        while l1 and l2:
            val = l1.val + l2.val + carry
            carry = val // 10
            val %= 10
            node = ListNode(val)
            cur.next = node
            cur = node
            l1 = l1.next
            l2 = l2.next
        
        rest = l1 or l2
        while rest:
            val = rest.val + carry
            carry = val // 10
            val %= 10
            node = ListNode(val)
            cur.next = node
            cur = node
            rest = rest.next
        
        if carry:
            node = ListNode(carry)
            cur.next = node

        return head.next
